在执行阶段，你必须以 **"思考 → 行动 → 观察"** 的闭环方式推进任务：
- **思考（Thought）**：分析当前状态、下一步目标、可用工具及潜在障碍。
- **行动（Action）**：**必须调用以下工具之一**完成具体操作，禁止自行"假设"或"编造"数据。
- **观察（Observation）**：根据工具返回结果判断是否达成子目标，决定继续、回退或求助。

import asyncio
import contextlib
import json
import time
from collections.abc import AsyncGenerator

from camel.responses import ChatAgentResponse
from fastapi import Request
from loguru import logger
from sse_starlette import ServerSentEvent
from steins_agent.models import BaseBoxData, BaseEvent

async def react_callback(
    http_request: Request,
    request: KnowledgeAgentRequest,
    parser: ReactStreamOutputParser,
    agent_manager: KnowledgeAgentManager,
    snowflake_generator: SnowflakeGenerator,
    agent_task: asyncio.Task,
) -> AsyncGenerator[ServerSentEvent, None]:
    try:
        start_time = time.time() * 1000

        session_id = request.session_id
        message_id = request.message_id

        yield agent_manager.send_start_chat_node(
            message_id, request_body=request.model_dump()
        )
        node_ids = {}
        current_box_node_id = None
        message_task = None
        while True:
            message_task = asyncio.create_task(
                agent_manager.get_agent_message(session_id)
            )

            done, pending = await asyncio.wait(
                [agent_task, message_task], return_when=asyncio.FIRST_COMPLETED
            )

            if agent_task in done:
                # Check if message is also done (Race condition fix)
                if message_task in pending:
                    try:
                        await asyncio.wait_for(message_task, timeout=5.0)
                    except asyncio.CancelledError:
                        raise asyncio.CancelledError()
                    except Exception as e:
                        logger.error(f"Error waiting for message: {e}")
                        message_task.cancel()
                        with contextlib.suppress(asyncio.CancelledError):
                            await message_task

                if message_task.done() and not message_task.cancelled():
                    try:
                        delta_response_str = message_task.result()
                        delta_response = ChatAgentResponse(
                            **json.loads(delta_response_str)
                        )
                        if delta_response.terminated:
                            if current_box_node_id is not None:
                                box_data = BaseBoxData(title="Thinked", style="default")
                                yield agent_manager.send_box_node(
                                    message_id,
                                    box_data,
                                    BaseEvent.SEND_FINISH_BOX,
                                    current_box_node_id,
                                )
                                current_box_node_id = None

                            end_time = time.time() * 1000
                            time_cost = int(end_time - start_time)
                            yield agent_manager.send_end_chat_node(
                                message_id, time_cost=time_cost
                            )
                            return
                    except asyncio.CancelledError:
                        raise asyncio.CancelledError()
                    except Exception as e:
                        logger.error(f"Error processing message: {e}")

                if agent_task.cancelled():
                    raise asyncio.CancelledError()

                exc = agent_task.exception()
                if exc:
                    raise exc

                logger.warning(
                    f"Agent task {session_id} finished unexpectedly without termination signal."
                )
                if current_box_node_id is not None:
                    box_data = BaseBoxData(title="Thinked", style="default")
                    yield agent_manager.send_box_node(
                        message_id,
                        box_data,
                        BaseEvent.SEND_FINISH_BOX,
                        current_box_node_id,
                    )
                    current_box_node_id = None

                end_time = time.time() * 1000
                time_cost = int(end_time - start_time)
                yield agent_manager.send_end_chat_node(message_id, time_cost=time_cost)
                return

            delta_response_str = message_task.result()

            delta_response = ChatAgentResponse(**json.loads(delta_response_str))

            event = parser.parse(delta_response)

            for event_item in event:
                if event_item["type"] in [
                    BaseEvent.SEND_FINAL_CONTENT,
                    BaseEvent.SEND_THOUGHT_CONTENT,
                ]:
                    event_item = parser.handle_special_char(event_item)
                if not event_item:
                    continue
                if event_item["type"] == BaseEvent.SEND_THOUGHT_CONTENT:
                    if not current_box_node_id:
                        current_box_node_id = snowflake_generator.next()
                        box_data = BaseBoxData(title="Thinking", style="default")
                        yield agent_manager.send_box_node(
                            message_id,
                            box_data,
                            BaseEvent.SEND_CREATE_BOX,
                            current_box_node_id,
                        )

                    if event_item["type"] not in node_ids:
                        node_ids[event_item["type"]] = snowflake_generator.next()

                    yield agent_manager.send_chat_node(
                        message_id,
                        event_item["content"],
                        event_item["type"],
                        node_ids[event_item["type"]],
                        box_id=current_box_node_id,
                    )
                    # logger.debug(f"[Thinking]: {event_item['content']}")

                elif event_item["type"] == BaseEvent.SEND_TOOL_CALL:
                    node_ids.clear()
                    if not current_box_node_id:
                        current_box_node_id = snowflake_generator.next()
                        box_data = BaseBoxData(title="Thinking", style="default")
                        yield agent_manager.send_box_node(
                            message_id,
                            box_data,
                            BaseEvent.SEND_CREATE_BOX,
                            current_box_node_id,
                        )

                    if event_item["tool_name"] == "rag_search":
                        node_ids[event_item["type"]] = snowflake_generator.next()
                        tool_name = event_item["tool_name"]
                        tool_args = event_item["tool_args"]
                        raw_tool_result = event_item["tool_result"]

                        rag_search_data: RAGSearchData
                        if isinstance(raw_tool_result, str):
                            try:
                                tool_result_dict = json.loads(raw_tool_result)
                                rag_search_data = RAGSearchData(**tool_result_dict)
                            except json.JSONDecodeError:
                                logger.error(
                                    f"Failed to parse tool_result as JSON: {raw_tool_result}"
                                )
                                continue
                        else:
                            logger.error(
                                f"Unsupported tool_result type: {type(raw_tool_result)}"
                            )
                            continue

                        tool_time_cost = rag_search_data.time_cost
                        tool_status = rag_search_data.status
                        tool_result = rag_search_data.tool_result or []

                        if tool_status == "failed":
                            if rag_search_data.error_msg:
                                error_msg = rag_search_data.error_msg
                            else:
                                error_msg = "RAG internal unknown error"
                            logger.error(
                                f"Tool {tool_name} execution failed: {error_msg}"
                            )
                            continue

                        parsed_result = parse_rag_search_result(
                            tool_name=tool_name,
                            tool_args=tool_args,
                            tool_result=tool_result,
                        )
                        yield agent_manager.send_tool_call_node(
                            message_id,
                            parsed_result,
                            node_ids[event_item["type"]],
                            time_cost=tool_time_cost,
                            box_id=current_box_node_id,
                        )
                        logger.debug(f"[RAG Search]: \n{parsed_result}")

                    if event_item["tool_name"] == "get_current_time":
                        logger.info(f"[Current Time]: {event_item['tool_result']}")

                elif event_item["type"] == BaseEvent.SEND_FINAL_CONTENT:
                    if current_box_node_id is not None:
                        box_data = BaseBoxData(title="Thinked", style="default")
                        yield agent_manager.send_box_node(
                            message_id,
                            box_data,
                            BaseEvent.SEND_FINISH_BOX,
                            current_box_node_id,
                        )
                        current_box_node_id = None

                    if event_item["type"] not in node_ids:
                        node_ids[event_item["type"]] = snowflake_generator.next()

                    yield agent_manager.send_chat_node(
                        message_id,
                        event_item["content"],
                        event_item["type"],
                        node_ids[event_item["type"]],
                    )
                    logger.debug(f"[Final Answer]: {event_item['content']}")

            if delta_response.terminated:
                if current_box_node_id is not None:
                    box_data = BaseBoxData(title="Thinked", style="default")
                    yield agent_manager.send_box_node(
                        message_id,
                        box_data,
                        BaseEvent.SEND_FINISH_BOX,
                        current_box_node_id,
                    )
                    current_box_node_id = None

                end_time = time.time() * 1000
                time_cost = int(end_time - start_time)
                yield agent_manager.send_end_chat_node(message_id, time_cost=time_cost)
                return

    except asyncio.CancelledError:
        logger.debug(f"react_callback cancelled for session {session_id}")
        if message_task and not message_task.done():
            message_task.cancel()
        if agent_task and not agent_task.done():
            agent_task.cancel()
        raise

    except Exception as e:
        end_time = time.time() * 1000
        time_cost = int(end_time - start_time)
        logger.error(f"react_callback error: {e}")
        if current_box_node_id is not None:
            box_data = BaseBoxData(title="Thinked", style="default")
            yield agent_manager.send_box_node(
                message_id, box_data, BaseEvent.SEND_FINISH_BOX, current_box_node_id
            )
        yield agent_manager.send_error_chat_node(
            message_id, str(e), time_cost=time_cost
        )
        yield agent_manager.send_end_chat_node(message_id, time_cost=time_cost)
        return
    finally:
        if not agent_task.done():
            agent_task.cancel()
            logger.info(f"Agent task for {session_id} cancelled in finally block.")
        if message_task and not message_task.done():
            message_task.cancel()
            logger.info(f"Message task for {session_id} cancelled in finally block.")


async def common_callback(
    http_request: Request,
    request: KnowledgeAgentRequest,
    parser: CommonStreamOutputParser,
    agent_manager: KnowledgeAgentManager,
    snowflake_generator: SnowflakeGenerator,
    agent_task: asyncio.Task,
) -> AsyncGenerator[ServerSentEvent, None]:
    try:
        start_time = time.time() * 1000

        session_id = request.session_id
        message_id = request.message_id

        yield agent_manager.send_start_chat_node(
            message_id, request_body=request.model_dump()
        )
        node_ids = {}
        current_box_node_id = None
        have_send_tool_call = False
        message_task = None
        while True:
            # Create a task for getting the message
            message_task = asyncio.create_task(
                agent_manager.get_agent_message(session_id)
            )

            # Wait for either the agent task to finish (crash) or a message to arrive
            done, pending = await asyncio.wait(
                [agent_task, message_task], return_when=asyncio.FIRST_COMPLETED
            )

            # If agent_task is done, it means it finished or crashed
            if agent_task in done:
                # Check if message is also done (Race condition fix)
                if message_task in pending:
                    try:
                        await asyncio.wait_for(message_task, timeout=5.0)
                    except Exception:
                        message_task.cancel()
                        with contextlib.suppress(asyncio.CancelledError):
                            await message_task

                if message_task.done() and not message_task.cancelled():
                    try:
                        delta_response_str = message_task.result()
                        delta_response = ChatAgentResponse(
                            **json.loads(delta_response_str)
                        )
                        if delta_response.terminated:
                            if current_box_node_id is not None:
                                box_data = BaseBoxData(
                                    title="Tool Calls", style="default"
                                )
                                yield agent_manager.send_box_node(
                                    message_id,
                                    box_data,
                                    BaseEvent.SEND_FINISH_BOX,
                                    current_box_node_id,
                                )
                                current_box_node_id = None

                            end_time = time.time() * 1000
                            time_cost = int(end_time - start_time)
                            yield agent_manager.send_end_chat_node(
                                message_id, time_cost=time_cost
                            )
                            return
                    except Exception as e:
                        logger.error(f"Error processing message: {e}")

                if agent_task.cancelled():
                    raise asyncio.CancelledError()

                exc = agent_task.exception()
                if exc:
                    raise exc

                logger.warning(
                    f"Agent task {session_id} finished unexpectedly without termination signal."
                )
                if current_box_node_id is not None:
                    box_data = BaseBoxData(title="Tool Calls", style="default")
                    yield agent_manager.send_box_node(
                        message_id,
                        box_data,
                        BaseEvent.SEND_FINISH_BOX,
                        current_box_node_id,
                    )
                    current_box_node_id = None

                end_time = time.time() * 1000
                time_cost = int(end_time - start_time)
                yield agent_manager.send_end_chat_node(message_id, time_cost=time_cost)
                return

            # If we are here, message_task is done
            delta_response_str = message_task.result()
            delta_response = ChatAgentResponse(**json.loads(delta_response_str))

            if not have_send_tool_call and (
                tool_calls := delta_response.info.get("tool_calls")
            ):
                if not current_box_node_id:
                    current_box_node_id = snowflake_generator.next()
                    box_data = BaseBoxData(title="Tool Calls", style="default")
                    yield agent_manager.send_box_node(
                        message_id,
                        box_data,
                        BaseEvent.SEND_CREATE_BOX,
                        current_box_node_id,
                    )

                for tool_call in tool_calls:
                    tool_call_dict = tool_call

                    tool_call = (
                        tool_call_dict
                        if isinstance(tool_call_dict, dict)
                        else tool_call_dict.__dict__
                    )

                    raw_tool_result = tool_call_dict.get("result")

                    rag_search_data: RAGSearchData
                    if isinstance(raw_tool_result, str):
                        try:
                            tool_result_dict = json.loads(raw_tool_result)
                            rag_search_data = RAGSearchData(**tool_result_dict)
                        except json.JSONDecodeError:
                            logger.error(
                                f"Failed to parse tool_result as JSON: {raw_tool_result}"
                            )
                            continue
                    else:
                        logger.error(
                            f"Unsupported t_result type: {type(raw_tool_result)}"
                        )
                        continue

                    tool_node_id = snowflake_generator.next()
                    tool_time_cost = rag_search_data.time_cost
                    tool_status = rag_search_data.status
                    tool_result = rag_search_data.tool_result or []

                    if tool_status == "failed":
                        error_msg = "Unknown error"
                        if (
                            tool_result
                            and isinstance(tool_result, list)
                            and len(tool_result) > 0
                        ):
                            error_msg = tool_result[0].get("error", "Unknown error")
                        logger.error(f"Tool rag_search failed: {error_msg}")
                        continue

                    parsed_result = parse_rag_search_result(
                        tool_name=tool_call_dict.get("name", "rag_search"),
                        tool_args=tool_call_dict.get("arguments", {}),
                        tool_result=tool_result,
                    )
                    yield agent_manager.send_tool_call_node(
                        message_id,
                        parsed_result,
                        tool_node_id,
                        time_cost=tool_time_cost,
                        box_id=current_box_node_id,
                    )
                    logger.debug(f"[RAG Search]: \n{parsed_result}")
                have_send_tool_call = True

            if delta_response.terminated:
                if current_box_node_id is not None:
                    box_data = BaseBoxData(title="Tool Calls", style="default")
                    yield agent_manager.send_box_node(
                        message_id,
                        box_data,
                        BaseEvent.SEND_FINISH_BOX,
                        current_box_node_id,
                    )
                    current_box_node_id = None

                end_time = time.time() * 1000
                time_cost = int(end_time - start_time)
                yield agent_manager.send_end_chat_node(message_id, time_cost=time_cost)
                return

            events = parser.parse(delta_response)

            for event in events:
                if event["type"] == BaseEvent.SEND_FINAL_CONTENT:
                    if current_box_node_id is not None:
                        box_data = BaseBoxData(title="Tool Calls", style="default")
                        yield agent_manager.send_box_node(

                            box_data,
                            BaseEvent.SEND_FINISH_BOX,
                            current_box_node_id,
                        )

                    yield agent_manager.send_chat_node(
                        message_id,
                        event["content"],
                        event["type"],
                        node_ids[event["type"]],
                        box_id=current_box_node_id,

    except asyncio.CancelledError:

        logger.error(f"common_callback error: {e}")
        if current_box_node_id is not None:
            box_data = BaseBoxData(title="Tool Calls", style="default")
            yield agent_manager.send_box_node(
                message_id, box_data, BaseEvent.SEND_FINISH_BOX, current_box_node_id

            current_box_node_id = None
        yield agent_manager.send_error_chat_node(
            message_id, str(e), time_cost=time_cost
        )
        yield agent_manager.send_end_chat_node(message_id, time_cost=time_cost)
        return
