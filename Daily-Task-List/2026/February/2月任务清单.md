```
async def arun(self, agent_id: str, message: str, agent: SingleAgent) -> None:

        """

        运行对话方法，将模型产生的消息推送到消息队列中

  

        Args:

            agent_id (str): id

            message (str): 输入消息

            agent (SingleAgen): 被用于运行的agent

        """

        try:

            async for response in agent.arun(message):

                dump_content = response.model_dump_json(

                    ensure_ascii=False, serialize_as_any=True

                )

  

                await self.message_queue.push_message(

                    agent_id,

                    dump_content,

                )

  

        except Exception as e:

            error_msg = f"AsyncAgentEngine | arun 方法操作发生非预期错误：| {str(e)}"

            await self.message_queue.push_message(

                agent_id,

                error_msg,

            )

            raise SteinsAgentException(error_msg) from e
```
